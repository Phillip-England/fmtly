func SurvivalSupplies(members []Memeber) string {
        var builder strings.Builder
        memberFor := gtml.For(members, func(i int, member Memeber) string {
                var memberBuilder strings.Builder
                survivallistPlaceholder := func() string {
                        return SurvivalList(member.Name)
                }
                memberBuilder.WriteString(`<ul _for="member of members []Memeber">`)
                memberBuilder.WriteString(survivallistPlaceholder())
                memberBuilder.WriteString(`</ul>`)
                return memberBuilder.String()
        })
        builder.WriteString(`<div _component="SurvivalSupplies">`)
        builder.WriteString(memberFor)
        builder.WriteString(`</div>`)
        return builder.String()
}
