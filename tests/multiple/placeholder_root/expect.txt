func GreetingCard(guestName string, message string) string {
        var builder strings.Builder
        builder.WriteString(`<div _component="GreetingCard" _id="0"><h1>Hello, `)
        builder.WriteString(guestName)
        builder.WriteString(`! How are you?</h1><p>I&#39;ve been waiting to tell you.. `)
        builder.WriteString(message)
        builder.WriteString(`</p></div>`)
        return builder.String()
}
