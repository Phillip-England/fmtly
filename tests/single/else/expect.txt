func UserGate(isBlocked bool) string {
	var builder strings.Builder
	isBlockedElse := gtml.Else(isBlocked, func() string {
		var isBlockedBuilder strings.Builder
		isBlockedBuilder.WriteString(`<div _else="isBlocked"><p>you are not blocked!</p></div>`)
		if !isBlocked {
			return isBlockedBuilder.String()
		}
		return ""
	})
	builder.WriteString(`<div _component="UserGate">`)
	builder.WriteString(isBlockedElse)
	builder.WriteString(`</div>`)
	return builder.String()
}
