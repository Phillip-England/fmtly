func UserGate(warning string, isBlocked bool) string {
        var builder strings.Builder
        isBlockedElse1 := gtml.Else(isBlocked, func() string {
                var isBlockedBuilder strings.Builder
                isBlockedBuilder.WriteString(`<div _else="isBlocked" _id="1"><p>you are not blocked!</p>`)
                isBlockedBuilder.WriteString(warning)
                isBlockedBuilder.WriteString(`</div>`)
                if !isBlocked {
                        return isBlockedBuilder.String()
                }
                return ""
        })
        builder.WriteString(`<div _component="UserGate" _id="0">`)
        builder.WriteString(isBlockedElse1)
        builder.WriteString(`</div>`)
        return builder.String()
}
